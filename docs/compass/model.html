  37def read_households(path: str) -> pd.DataFrame:
  38    """
  39    A fast and efficient parser to read household data from a GeoJSON file.
  40    Arguments
  41        path    File to open, passed directly to open()
  42    Returns
  43        households (pd.DataFrame) [pos, category, neighbourhood_id]
  44    """
  45
  46    # open in binary mode for ijson
  47    geofile = open(path, "rb")
  48
  49    # assume floats are good enough,
  50    # no need for aribrary precision Decimal objects
  51    geo_households = ijson.items(geofile, "features.item", use_float=True)
  52
  53    # normal python lists, because appending is fast for them,
  54    # and we need the columns to create a pandas DataFrame
  55    coordinates, groups, neighbourhood_ids = [], [], []
  56    
  57    for feature in geo_households:
  58        coordinates.append(feature["geometry"]["coordinates"])
  59        groups.append(feature["properties"]["group"])
  60        neighbourhood_ids.append(feature["properties"]["neighbourhood_id"])
  61    geofile.close()
  62
  63    return pd.DataFrame(
  64        {"pos": coordinates, "category": groups, "neighbourhood_id": neighbourhood_ids}
  65    )

  68def trunc_normal_sample(means: List, scale: float, size: int, seed=None) -> List[np.ndarray]:
  69    """
  70    Samples from a set of truncated normal distributions.
  71    Arguments:
  72        means  [n_means] of float
  73        scale  float
  74        size   float
  75    Output:
  76        sample [n_means] of np.ndarray [size]
  77    """
  78    truncnorm._random_state = seed
  79
  80    sample = [0] * len(means)
  81    for index, mu in enumerate(means):
  82        if scale == 0:
  83            # All samples are equal if the scale is zero.
  84            sample[index] = np.ones(size) * mu
  85        else:
  86            sample[index] = truncnorm.rvs(
  87                (0 - mu) / scale, (1 - mu) / scale, loc=mu, scale=scale, size=size
  88            )
  89    return sample

 281    def set_agent_parameters(self, params: dict) -> None:
 282        """
 283        Puts the agent parameters in numpy arrays for faster computations.
 284        Args:
 285            params (dict): Model parameters which could differ from
 286                the agent params!
 287        Todo:
 288            Parameters should be imported from a config file in the future.
 289        """
 290        households = self.get_agents("households")
 291        n_households = len(households)
 292
 293        dtype = "float32"
 294        self.p = np.zeros(n_households, dtype=dtype)
 295        self.q = np.zeros(n_households, dtype=dtype)
 296        self.alpha = np.zeros(n_households, dtype=dtype)
 297        self.temperature = self.params["temperature"]
 298        self.utility_at_max = np.zeros(n_households, dtype=dtype)
 299        self.optimal_fraction = np.zeros(n_households, dtype=dtype)
 300        self.neighbourhood_mixture = np.ones(n_households, dtype=int)
 301
 302        num_types = len(self.params["group_types"][0])
 303        self.num_types = num_types # for use in visualisation
 304        
 305        # If every parameter of a subset (see check_num_params) has enough
 306        # values for all types, use these. If not, give them all the first one
 307        if self.check_num_params(num_types):
 308            optimal_fractions = np.tile(params["optimal_fraction"][0], (n_households, 1)).T
 309            alphas = np.tile(params["alpha"][0], (n_households, 1)).T
 310            utility_at_maxs = np.tile(params["utility_at_max"][0], (n_households, 1)).T
 311            ps = np.tile(params["p"][0], (n_households, 1)).T
 312            qs = np.tile(params["q"][0], (n_households, 1)).T
 313
 314        else:
 315            optimal_fractions,
 316            alphas, utility_at_maxs,
 317            ps, qs = [trunc_normal_sample([params[par][0][0]]*num_types,
 318                                            scale=params["stdev"],
 319                                            size=n_households,
 320                                            seed=self.random
 321                        ) for par in ['optimal_fraction', 'alpha', 
 322                        'utility_at_max', 'p', 'q']]
 323
 324        # Set parameters of every household
 325        for household in households:
 326            x, y = household.pos
 327
 328            # Fill arrays with agent parameter values for faster computations
 329            self.optimal_fraction[household.idx] = optimal_fractions[
 330                household.category
 331            ][household.idx]
 332            self.alpha[household.idx] = alphas[household.category][household.idx]
 333            self.utility_at_max[household.idx] = utility_at_maxs[household.category][
 334                household.idx
 335            ]
 336            self.p[household.idx] = ps[household.category][household.idx]
 337            self.q[household.idx] = qs[household.category][household.idx]
 338
 339            # Currently only convolution (assumes every household has the same
 340            # radius) for composition calculations within the lattice case.
 341            if params["case"].lower() == "lattice":
 342                self.local_compositions.append(
 343                    self.normalized_compositions[x, y, household.category]
 344                )
 345            else:
 346                household.params["neighbourhood_mixture"] = 1
 347
 348            if household.neighbourhood.total > 0:
 349                norm = 1.0 / household.neighbourhood.total
 350            else:
 351                norm = 1.0
 352            self.neighbourhood_compositions.append(
 353                household.neighbourhood.composition[household.category] * norm
 354            )
 355
 356        self.school_compositions = np.zeros(n_households, dtype=dtype)
 357
 358        # Distance utilities based on sigmoid function
 359        if self.params["case"].lower() != "lattice":
 360            
 361            self.distance_utilities = 1.0 / (1 + (self.all_distances / self.p[:,np.newaxis]) ** self.q[:,np.newaxis])
 362
 363            if self.params['subset_schools']:
 364                N = self.params['num_considered']
 365                newval = 0
 366                self.distance_utilities = np.ones(self.all_distances.shape)
 367                np.put_along_axis(
 368                    self.distance_utilities, 
 369                    np.argpartition(self.all_distances, N, axis=1)[:, N:], 
 370                    newval, axis=1)

 587    def create_households(
 588        self, household_frame: pd.DataFrame, actual_households: int
 589    ) -> None:
 590        """
 591        Given a GeoDataFrame, this creates household objects by randomly
 592        drawing (without replacement) household metadata rows from the frame.
 593        Arguments
 594            actual_households (int)
 595                The number of households to create.
 596        Parameters used
 597            params['random_residential']
 598        Parameters set
 599            params['n_households']
 600        """
 601        if self.verbose:
 602            print("Creating households...")
 603
 604        self.chosen_indices = self.random.choice(
 605            len(household_frame), size=actual_households, replace=False
 606        )
 607        households_sample = household_frame.iloc[self.chosen_indices]

 654    def calc_residential_compositions(self) -> None:
 655        """
 656        Updates all local residential compositions assuming all households have

 658        """
 659
 660        # Determine the kernel of the convolution
 661        radius = self.params["radius"]
 662        dim = radius * 2 + 1
 663        self.kernel = np.ones((dim, dim))
 664        self.kernel[radius, radius] = 0

 666        # Should it wrap around the edges or not?

 669        else:
 670            mode = "constant"

 672        summed = 0

 675
 676        # Convolution for every household attribute.
 677        for attr in range(num_attrs):
 678            compositions[:, :, attr] = convolve(
 679                self.household_attrs[:, :, attr], self.kernel, mode=mode

 681            summed += compositions[:, :, attr]

 684            compositions / np.repeat(summed[:, :, np.newaxis], num_attrs, axis=2)

 744        self, households: List[Household], schools: List[School]
 745    ) -> None:
 746        """
 747        Ranks the schools according to utility.
 748        Note: this is probably a performance bottleneck; one of the array is
 749        of size [n_households_moving, n_of_schools].
 750        Args:
 751            households (list): list of households the rankings need to be
 752                calculated for.
 753            schools (list): list of schools that need to be ranked.
 754        Todo:
 755            Schools can differ per household if we only want to look at the
 756            n-closest schools for example?
 757        """
 758        if len(schools) == 0 or len(households) == 0:
 759            return
 760
 761        zeros = np.zeros(len(self.params["group_types"][0]))
 762        compositions = np.array(
 763            [
 764                school.composition / school.total if school.total > 0 else zeros
 765                for school in schools
 766            ],
 767            dtype="float32",
 768        )
 769
 770        households_data = np.array(
 771            [(h.idx, h.category) for h in households], dtype=int
 772        ).T
 773        households_indices = households_data[0, :]
 774        households_categories = households_data[1, :]
 775
 776        # Composition utility calculations
 777        t = np.take(self.optimal_fraction, households_indices)  # [n_indices]
 778        M = np.take(self.utility_at_max, households_indices)  # [n_indices]
 779        x = np.take(
 780            compositions, households_categories, axis=1
 781        )  # [n_schools, n_indices]
 782        composition_utilities = np.where(x <= t, x / t, M + (1 - x) * (1 - M) / (1 - t))
 783
 784        # Combined (THIS SHOULD BE GENERALISED TO INCLUDE MORE FACTORS)
 785        utilities = (
 786            composition_utilities * self.alpha[np.newaxis, households_indices]
 787            + (
 788                np.take(self.distance_utilities, households_indices, axis=0)
 789                * (1 - self.alpha[households_indices, np.newaxis])
 790            ).T
 791        )
 792        
 793        households_utilities = np.take(
 794            Household._household_school_utility, households_indices
 795        )
 796
 797        method = self.params["ranking_method"].lower()
 798        if method == "proportional":
 799            differences = utilities - households_utilities[np.newaxis, :]
 800            exp_utilities = np.exp(self.temperature * differences)
 801            transformed = exp_utilities / exp_utilities.sum(axis=0)[np.newaxis, :]
 802
 803        elif method == "highest":
 804            transformed = utilities
 805
 806        else:
 807            print("Method not implementd.")
 808            sys.exit()
 809
 810        # instead of reversing the list, sort the negative values in the list
 811        ranked_indices = (-transformed).argsort(axis=0, kind='quicksort')
 812
 813        # necessary to allow indexing with the argsort result
 814        schools = np.array(schools)
 815
 816        for i, household in enumerate(households):
 817            ranking = schools[ranked_indices[:, i]]

 924    def convergence_check(self) -> bool:
 925        """
 926        Checks if the processes have converged.
 927        Returns: True if converged.
 928        """
 929        window_size = self.params["window_size"]
 930        time = self.scheduler.get_time()
 931        school_time = self.scheduler.get_time("school")
 932
 933        # Check what type of segregation to calculate (i.e., which of the
 934        # processes is running)
 935        if not self.res_ended:
 936            self.segregation.append(
 937                self.measurements.calculate_segregation(
 938                    type="bounded_neighbourhood", index="Theil"
 939                )
 940            )
 941        else:
 942            self.segregation.append(
 943                self.measurements.calculate_segregation(type="school", index="Theil")
 944            )
 945
 946        # Wait until there is enough steps in the school process
 947        if (self.res_ended and school_time <= window_size):
 948            return False
 949
 950        # Check all metrics in the window size and check if they are below
 951        # the convergence threshold
 952        if time >= window_size:
 953            utilities = self.measurements.households[
 954                time - window_size + 1 : time + 1, :, 2
 955            ]
 956            means = utilities.mean(axis=1)
 957            stds = utilities.std(axis=1)
 958
 959            metrics = np.vstack(
 960                (means, stds, self.segregation[time - window_size + 1 : time + 1])
 961            )
 962
 963            metric_means = np.repeat(
 964                metrics.mean(axis=1)[:, np.newaxis], window_size, axis=1
 965            )
 966            mad = np.abs(metrics - metric_means)
 967            if np.all(mad < self.params["conv_threshold"]):
 968                # Start over if the residential process has converged
 969                self.res_ended = True
 970                return True
 971
 972        return False

 974    def choose_locations(
 975        self, amount: int, method: str = "evenly_spaced"
 976    ) -> List[tuple[float, float]]:
 977        """
 978        Compute a number of locations to place school and neighbourhood objects.
 979        Args:
 980            amount (int): the number of agents to place.
 981            method (str): one of 'evenly_spaced', 'random', 'random_per_neighbourhood'
 982        Returns:
 983            list: containing all the locations in tuple (x,y) format.
 984        """
 985
 986        if amount == 0:
 987            return []
 988
 989        if method == "evenly_spaced":
 990            per_side = np.sqrt(amount)
 991            if per_side % 1 != 0:
 992                print("Unable to place amount of locations using given method")
 993                sys.exit(1)
 994
 995            # Compute locations
 996            per_side = int(per_side)
 997            xs = np.linspace(0, self.params["width"], per_side * 2 + 1)[1::2]
 998            ys = np.linspace(0, self.params["height"], per_side * 2 + 1)[1::2]
 999            return [(x, y) for x in xs for y in ys]
1000
1001        elif method == "random":
1002            locations = []
1003            i = 0
1004            while i < amount:
1005                x_coord = self.random.randint(low=0, high=self.params["width"])
1006                y_coord = self.random.randint(low=0, high=self.params["height"])
1007                if (x_coord, y_coord) not in locations:
1008                    locations.append((x_coord, y_coord))
1009                    i += 1
1010            return locations
1011
1012        elif method == "random_per_neighbourhood":
1013            locations = []
1014            width, height = self.params["width"], self.params["height"]
1015            n_schools = self.params["n_schools"]
1016            n_neighbourhoods = self.params["n_neighbourhoods"]
1017            per_side = int(np.sqrt(n_neighbourhoods))
1018            location_width = width / per_side
1019            location_height = height / per_side
1020
1021            # Draw a random sample per neighbourhood as long as there are
1022            # schools to place
1023            i = 0
1024            while i < max(n_neighbourhoods, n_schools):
1025                y_low = 0
1026                for col in range(per_side):
1027                    x_low = 0
1028                    y_high = int((1 + col) * location_height)
1029
1030                    for row in range(per_side):
1031
1032                        x_high = int((1 + row) * location_width)
1033
1034                        if x_high >= width:
1035                            x_high = width - 1
1036                        elif y_high >= height:
1037                            y_high = height - 1
1038
1039                        x_coord = self.random.randint(low=x_low, high=x_high)
1040                        y_coord = self.random.randint(low=y_low, high=y_high)
1041
1042                        # Check if the coordinates haven't already been sampled
1043                        while (x_coord, y_coord) in locations:
1044                            x_coord = self.random.randint(low=x_low, high=x_high)
1045                            y_coord = self.random.randint(low=y_low, high=y_high)
1046
1047                        locations.append((x_coord, y_coord))
1048                        x_low = x_high + 1
1049                        i += 1
1050                    y_low = y_high + 1
1051
1052            # Shuffle all locations if n_schools <= n_neighbourhoods, otherwise
1053            # shuffle only the remainder
1054            if n_schools <= n_neighbourhoods:
1055                self.random.shuffle(locations)
1056            else:
1057                divider = int(n_schools / n_neighbourhoods)
1058                remainder = n_schools % n_neighbourhoods
1059                first_locations = locations[: n_neighbourhoods * divider]
1060                rest_locations = locations[n_neighbourhoods * divider :]
1061                self.random.shuffle(rest_locations)
1062                locations = first_locations + rest_locations[:remainder]
1063
1064            return locations
1065
1066        print(f"Unknown method in choose_locations {method} __file__:__line__")
1067        sys.exit(-1)

 282    def set_agent_parameters(self, params: dict) -> None:
 283        """
 284        Puts the agent parameters in numpy arrays for faster computations.
 285        Args:
 286            params (dict): Model parameters which could differ from
 287                the agent params!
 288        Todo:
 289            Parameters should be imported from a config file in the future.
 290        """
 291        households = self.get_agents("households")
 292        n_households = len(households)
 293
 294        dtype = "float32"
 295        self.p = np.zeros(n_households, dtype=dtype)
 296        self.q = np.zeros(n_households, dtype=dtype)
 297        self.alpha = np.zeros(n_households, dtype=dtype)
 298        self.temperature = self.params["temperature"]
 299        self.utility_at_max = np.zeros(n_households, dtype=dtype)
 300        self.optimal_fraction = np.zeros(n_households, dtype=dtype)
 301        self.neighbourhood_mixture = np.ones(n_households, dtype=int)
 302
 303        num_types = len(self.params["group_types"][0])
 304        self.num_types = num_types # for use in visualisation
 305        
 306        # If every parameter of a subset (see check_num_params) has enough
 307        # values for all types, use these. If not, give them all the first one
 308        if self.check_num_params(num_types):
 309            optimal_fractions = np.tile(params["optimal_fraction"][0], (n_households, 1)).T
 310            alphas = np.tile(params["alpha"][0], (n_households, 1)).T
 311            utility_at_maxs = np.tile(params["utility_at_max"][0], (n_households, 1)).T
 312            ps = np.tile(params["p"][0], (n_households, 1)).T
 313            qs = np.tile(params["q"][0], (n_households, 1)).T
 314
 315        else:
 316            optimal_fractions,
 317            alphas, utility_at_maxs,
 318            ps, qs = [trunc_normal_sample([params[par][0][0]]*num_types,
 319                                            scale=params["stdev"],
 320                                            size=n_households,
 321                                            seed=self.random
 322                        ) for par in ['optimal_fraction', 'alpha', 
 323                        'utility_at_max', 'p', 'q']]
 324
 325        # Set parameters of every household
 326        for household in households:
 327            x, y = household.pos
 328
 329            # Fill arrays with agent parameter values for faster computations
 330            self.optimal_fraction[household.idx] = optimal_fractions[
 331                household.category
 332            ][household.idx]
 333            self.alpha[household.idx] = alphas[household.category][household.idx]
 334            self.utility_at_max[household.idx] = utility_at_maxs[household.category][
 335                household.idx
 336            ]
 337            self.p[household.idx] = ps[household.category][household.idx]
 338            self.q[household.idx] = qs[household.category][household.idx]
 339
 340            # Currently only convolution (assumes every household has the same
 341            # radius) for composition calculations within the lattice case.
 342            if params["case"].lower() == "lattice":
 343                self.local_compositions.append(
 344                    self.normalized_compositions[x, y, household.category]
 345                )
 346            else:
 347                household.params["neighbourhood_mixture"] = 1
 348
 349            if household.neighbourhood.total > 0:
 350                norm = 1.0 / household.neighbourhood.total
 351            else:
 352                norm = 1.0
 353            self.neighbourhood_compositions.append(
 354                household.neighbourhood.composition[household.category] * norm
 355            )
 356
 357        self.school_compositions = np.zeros(n_households, dtype=dtype)
 358
 359        # Distance utilities based on sigmoid function
 360        if self.params["case"].lower() != "lattice":
 361            
 362            self.distance_utilities = 1.0 / (1 + (self.all_distances / self.p[:,np.newaxis]) ** self.q[:,np.newaxis])
 363
 364            if self.params['subset_schools']:
 365                N = self.params['num_considered']
 366                newval = 0
 367                self.distance_utilities = np.ones(self.all_distances.shape)
 368                np.put_along_axis(
 369                    self.distance_utilities, 
 370                    np.argpartition(self.all_distances, N, axis=1)[:, N:], 
 371                    newval, axis=1)

 655    def calc_residential_compositions(self) -> None:
 656        """
 657        Updates all local residential compositions assuming all households have
 658        the SAME RADIUS.
 659        """
 660
 661        # Determine the kernel of the convolution
 662        radius = self.params["radius"]
 663        dim = radius * 2 + 1
 664        self.kernel = np.ones((dim, dim))
 665        self.kernel[radius, radius] = 0
 666
 667        # Should it wrap around the edges or not?
 668        if self.params["torus"]:
 669            mode = "wrap"
 670        else:
 671            mode = "constant"
 672
 673        summed = 0
 674        num_attrs = self.household_attrs.shape[2]
 675        compositions = np.zeros(shape=self.household_attrs.shape, dtype="float32")
 676
 677        # Convolution for every household attribute.
 678        for attr in range(num_attrs):
 679            compositions[:, :, attr] = convolve(
 680                self.household_attrs[:, :, attr], self.kernel, mode=mode
 681            )
 682            summed += compositions[:, :, attr]
 683        self.compositions = compositions
 684        self.normalized_compositions = np.nan_to_num(
 685            compositions / np.repeat(summed[:, :, np.newaxis], num_attrs, axis=2)
 686        )

 744    def calc_school_rankings(
 745        self, households: List[Household], schools: List[School]
 746    ) -> None:
 747        """
 748        Ranks the schools according to utility.
 749        Note: this is probably a performance bottleneck; one of the array is
 750        of size [n_households_moving, n_of_schools].
 751        Args:
 752            households (list): list of households the rankings need to be
 753                calculated for.
 754            schools (list): list of schools that need to be ranked.
 755        Todo:
 756            Schools can differ per household if we only want to look at the
 757            n-closest schools for example?
 758        """
 759        if len(schools) == 0 or len(households) == 0:
 760            return
 761
 762        zeros = np.zeros(len(self.params["group_types"][0]))
 763        compositions = np.array(
 764            [
 765                school.composition / school.total if school.total > 0 else zeros
 766                for school in schools
 767            ],
 768            dtype="float32",
 769        )

 925    def convergence_check(self) -> bool:
 926        """
 927        Checks if the processes have converged.
 928        Returns: True if converged.
 929        """
 930        window_size = self.params["window_size"]
 931        time = self.scheduler.get_time()
 932        school_time = self.scheduler.get_time("school")
 933
 934        # Check what type of segregation to calculate (i.e., which of the
 935        # processes is running)
 936        if not self.res_ended:
 937            self.segregation.append(
 938                self.measurements.calculate_segregation(
 939                    type="bounded_neighbourhood", index="Theil"
 940                )
 941            )
 942        else:
 943            self.segregation.append(
 944                self.measurements.calculate_segregation(type="school", index="Theil")
 945            )
 946
 947        # Wait until there is enough steps in the school process
 948        if (self.res_ended and school_time <= window_size):
 949            return False
 950
 951        # Check all metrics in the window size and check if they are below
 952        # the convergence threshold
 953        if time >= window_size:
 954            utilities = self.measurements.households[
 955                time - window_size + 1 : time + 1, :, 2
 956            ]
 957            means = utilities.mean(axis=1)
 958            stds = utilities.std(axis=1)
 959
 960            metrics = np.vstack(
 961                (means, stds, self.segregation[time - window_size + 1 : time + 1])
 962            )
 963
 964            metric_means = np.repeat(
 965                metrics.mean(axis=1)[:, np.newaxis], window_size, axis=1
 966            )
 967            mad = np.abs(metrics - metric_means)
 968            if np.all(mad < self.params["conv_threshold"]):
 969                # Start over if the residential process has converged
 970                self.res_ended = True
 971                return True
 972
 973        return False

 992            if per_side % 1 != 0:

 999            ys = np.linspace(0, self.params["height"], per_side * 2 + 1)[1::2]

1007                y_coord = self.random.randint(low=0, high=self.params["height"])
1008                if (x_coord, y_coord) not in locations:
1009                    locations.append((x_coord, y_coord))
1010                    i += 1
1011            return locations
1012
1013        elif method == "random_per_neighbourhood":
1014            locations = []
1015            width, height = self.params["width"], self.params["height"]
1016            n_schools = self.params["n_schools"]
1017            n_neighbourhoods = self.params["n_neighbourhoods"]
1018            per_side = int(np.sqrt(n_neighbourhoods))
1019            location_width = width / per_side
1020            location_height = height / per_side
1021
1022            # Draw a random sample per neighbourhood as long as there are
1023            # schools to place
1024            i = 0
1025            while i < max(n_neighbourhoods, n_schools):
1026                y_low = 0
1027                for col in range(per_side):
1028                    x_low = 0
1029                    y_high = int((1 + col) * location_height)

1046                            y_coord = self.random.randint(low=y_low, high=y_high)

1049                        x_low = x_high + 1
1050                        i += 1

1053            # Shuffle all locations if n_schools <= n_neighbourhoods, otherwise
1054            # shuffle only the remainder
1055            if n_schools <= n_neighbourhoods:
1056                self.random.shuffle(locations)
1057            else:

1059                remainder = n_schools % n_neighbourhoods
1060                first_locations = locations[: n_neighbourhoods * divider]
1061                rest_locations = locations[n_neighbourhoods * divider :]
1062                self.random.shuffle(rest_locations)

925    def convergence_check(self) -> bool:
926        """
927        Checks if the processes have converged.
928        Returns: True if converged.
929        """
930        window_size = self.params["window_size"]
931        time = self.scheduler.get_time()
932        school_time = self.scheduler.get_time("school")
933
934        # Check what type of segregation to calculate (i.e., which of the
935        # processes is running)
936        if not self.res_ended:
937            self.segregation.append(
938                self.measurements.calculate_segregation(
939                    type="bounded_neighbourhood", index="Theil"
940                )
941            )
942        else:
943            self.segregation.append(
944                self.measurements.calculate_segregation(type="school", index="Theil")
945            )
946
947        # Wait until there is enough steps in the school process
948        if (self.res_ended and school_time <= window_size):
949            return False
950
951        # Check all metrics in the window size and check if they are below
952        # the convergence threshold
953        if time >= window_size:
954            utilities = self.measurements.households[
955                time - window_size + 1 : time + 1, :, 2
956            ]
957            means = utilities.mean(axis=1)
958            stds = utilities.std(axis=1)
959
960            metrics = np.vstack(
961                (means, stds, self.segregation[time - window_size + 1 : time + 1])
962            )
963
964            metric_means = np.repeat(
965                metrics.mean(axis=1)[:, np.newaxis], window_size, axis=1
966            )
967            mad = np.abs(metrics - metric_means)
968            if np.all(mad < self.params["conv_threshold"]):
969                # Start over if the residential process has converged
970                self.res_ended = True
971                return True
972
973        return False
