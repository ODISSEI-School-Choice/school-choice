 285    def composition_data(self, agent_type='school'):
 286        """
 287        Calculates the composition data for schools or neighbourhoods.
 288
 289        Args:
 290            agent_type (str): either 'school' or 'neighbourhood'
 291
 292        Returns:
 293            DataFrame of fraction blues and reds per agent.
 294        """
 295
 296        # Loop over all schools to get the locations and compositions
 297        cols = ['group0', 'group1', 'dist_school']
 298        data = self.data[self.data.agent_type == agent_type][cols]
 299
 300        if agent_type == 'household':
 301            blues = data[data['group0'] == 1]['dist_school'].values
 302            reds = data[data['group1'] == 1]['dist_school'].values
 303            return {'blues': blues, 'reds': reds}
 304        else:
 305            totals = data.sum(axis=1).replace(0, 1).values
 306            fractions = data.div(totals, axis=0)
 307            blues = fractions['group0'].values
 308            reds = fractions['group1'].values
 309            return {'blues': blues, 'reds': reds}

 576    def step_button(self):
 577        """
 578        Checks which process need to be stepped and execute the step. The
 579        simulate function of the Model instance cannot be used as we need to
 580        visualise every step.
 581        """
 582        if self.run_button=='Stop':
 583            return
 584
 585        max_res_steps = self.params['max_res_steps']
 586        max_school_steps = self.params['max_school_steps']
 587        school_time = self.model.scheduler.get_time('school')
 588        residential_time = self.model.scheduler.get_time('residential')
 589
 590        # If residential is not converged yet or below max steps, do a step
 591        if (residential_time < max_res_steps and not self.model.res_ended):
 592            self.model.step(residential=True)
 593            self.model.res_ended = self.model.convergence_check()
 594        else:
 595            self.model.res_ended = True
 596
 597        # Initial school step needs to be executed
 598        if (school_time == 0 and self.model.res_ended):
 599            self.residential = False
 600            self.model.step(residential=False, initial_schools=True)
 601            self.model.convergence_check()
 602
 603        # Normal school steps
 604        elif (school_time < max_school_steps and self.model.res_ended and not self.model.school_ended):
 605            self.model.step(residential=False, initial_schools=False)
 606
 607        if self.model.convergence_check():
 608            self.model.school_ended = True
 609            print('School process ended.')
 610
 611        # Both processes are done/converged
 612        if (self.model.res_ended and self.model.school_ended):
 613            self.run_button()
 614            return
 615
 616        self.update_data()
 617

 286    def composition_data(self, agent_type='school'):
 287        """
 288        Calculates the composition data for schools or neighbourhoods.
 289
 290        Args:
 291            agent_type (str): either 'school' or 'neighbourhood'
 292
 293        Returns:
 294            DataFrame of fraction blues and reds per agent.

 298        cols = ['group0', 'group1', 'dist_school']

 300

 303            reds = data[data['group1'] == 1]['dist_school'].values

 308            blues = fractions['group0'].values
 309            reds = fractions['group1'].values

 577    def step_button(self):
 578        """
 579        Checks which process need to be stepped and execute the step. The
 580        simulate function of the Model instance cannot be used as we need to
 581        visualise every step.
 582        """
 583        if self.run_button=='Stop':
 584            return
 585
 586        max_res_steps = self.params['max_res_steps']
 587        max_school_steps = self.params['max_school_steps']
 588        school_time = self.model.scheduler.get_time('school')
 589        residential_time = self.model.scheduler.get_time('residential')
 590
 591        # If residential is not converged yet or below max steps, do a step
 592        if (residential_time < max_res_steps and not self.model.res_ended):
 593            self.model.step(residential=True)
 594            self.model.res_ended = self.model.convergence_check()
 595        else:
 596            self.model.res_ended = True
 597
 598        # Initial school step needs to be executed
 599        if (school_time == 0 and self.model.res_ended):
 600            self.residential = False
 601            self.model.step(residential=False, initial_schools=True)
 602            self.model.convergence_check()
 603
 604        # Normal school steps
 605        elif (school_time < max_school_steps and self.model.res_ended and not self.model.school_ended):
 606            self.model.step(residential=False, initial_schools=False)
 607
 608        if self.model.convergence_check():
 609            self.model.school_ended = True
 610            print('School process ended.')
 611
 612        # Both processes are done/converged
 613        if (self.model.res_ended and self.model.school_ended):
 614            self.run_button()
 615            return
 616
 617        self.update_data()
 618

597

599        if (school_time == 0 and self.model.res_ended):
600            self.residential = False
601            self.model.step(residential=False, initial_schools=True)
602            self.model.convergence_check()
603
604        # Normal school steps
605        elif (school_time < max_school_steps and self.model.res_ended and not self.model.school_ended):
606            self.model.step(residential=False, initial_schools=False)
607
608        if self.model.convergence_check():
609            self.model.school_ended = True
610            print('School process ended.')
611
612        # Both processes are done/converged
613        if (self.model.res_ended and self.model.school_ended):
614            self.run_button()
615            return
616
